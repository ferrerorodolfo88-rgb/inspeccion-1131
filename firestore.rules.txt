rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Función auxiliar para verificar autenticación
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Función para verificar si el usuario es admin
    function isAdmin() {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Función para verificar si es el propietario del recurso
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Reglas para la colección de registros
    match /registros/{registroId} {
      // Cualquier usuario autenticado puede leer registros
      allow read: if isAuthenticated();
      
      // Cualquier usuario autenticado puede crear registros
      allow create: if isAuthenticated() && 
                       request.resource.data.createdBy == request.auth.uid;
      
      // Solo el creador o un admin puede actualizar
      allow update: if isAuthenticated() && 
                       (resource.data.createdBy == request.auth.uid || isAdmin());
      
      // Solo el creador o un admin puede eliminar
      allow delete: if isAuthenticated() && 
                       (resource.data.createdBy == request.auth.uid || isAdmin());
    }
    
    // Reglas para la colección de usuarios
    match /usuarios/{userId} {
      // Cualquier usuario autenticado puede leer perfiles de otros
      allow read: if isAuthenticated();
      
      // Solo el propio usuario puede crear/actualizar su perfil
      allow create, update: if isOwner(userId);
      
      // Solo admins pueden eliminar usuarios
      allow delete: if isAdmin();
    }
    
    // Reglas para adjuntos (si se implementa Storage)
    match /adjuntos/{adjuntoId} {
      allow read: if isAuthenticated();
      allow write: if isAuthenticated();
    }
  }
}
